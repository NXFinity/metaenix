Rate Limits

Meta EN|IX API implements rate limiting to ensure fair usage and maintain service quality for all developers.

Overview​

Rate limits prevent abuse and ensure system stability. When you exceed a rate limit, you'll receive a 429 Too Many Requests response.

Rate Limit Headers​

All API responses include rate limit headers:

X-RateLimit-Limit: 5
X-RateLimit-Remaining: 3
X-RateLimit-Reset: 2025-01-15T10:15:00.000Z

Header Fields​

HeaderDescription
X-RateLimit-LimitMaximum number of requests allowed
X-RateLimit-RemainingNumber of requests remaining in current window
X-RateLimit-ResetISO 8601 timestamp when the limit resets

Reading Rate Limit Headers​

const response = await fetch('https://api.metaenix.com/v1/auth/login', {
  method: 'POST',
  body: JSON.stringify(credentials)
});

const limit = response.headers.get('X-RateLimit-Limit');
const remaining = response.headers.get('X-RateLimit-Remaining');
const reset = response.headers.get('X-RateLimit-Reset');

console.log(`Limit: ${limit}, Remaining: ${remaining}, Resets: ${reset}`);

Rate Limit Identification​

Rate limits are applied based on:

  1. User ID - For authenticated requests
  2. IP Address - For unauthenticated requests
  3. Application - For OAuth requests (future)

Handling Rate Limits​

Check Headers Before Requests​

async function makeRequest(url, options) {
  // Check remaining requests before making request
  const lastResponse = getLastResponse(url);
  if (lastResponse) {
    const remaining = parseInt(lastResponse.headers.get('X-RateLimit-Remaining') || '0');
    if (remaining === 0) {
      const reset = new Date(lastResponse.headers.get('X-RateLimit-Reset'));
      const waitTime = reset.getTime() - Date.now();
      if (waitTime > 0) {
        console.log(`Rate limit reached. Waiting ${waitTime}ms`);
        await new Promise(resolve => setTimeout(resolve, waitTime));
      }
    }
  }
  
  const response = await fetch(url, options);
  storeLastResponse(url, response);
  return response;
}

Handle 429 Responses​

async function fetchWithRateLimitHandling(url, options, maxRetries = 3) {
  for (let attempt = 0; attempt < maxRetries; attempt++) {
    const response = await fetch(url, options);
    
    if (response.status === 429) {
      const error = await response.json();
      const resetTime = new Date(error.resetAt);
      const waitTime = Math.max(0, resetTime.getTime() - Date.now());
      
      if (waitTime > 0 && attempt < maxRetries - 1) {
        console.log(`Rate limited. Waiting ${waitTime}ms`);
        await new Promise(resolve => setTimeout(resolve, waitTime));
        continue; // Retry after waiting
      }
      
      throw new Error('Rate limit exceeded');
    }
    
    return response;
  }
}

Exponential Backoff​

async function fetchWithBackoff(url, options, maxRetries = 3) {
  let delay = 1000; // Start with 1 second
  
  for (let attempt = 0; attempt < maxRetries; attempt++) {
    try {
      const response = await fetch(url, options);
      
      if (response.status === 429) {
        const error = await response.json();
        const resetTime = new Date(error.resetAt);
        const waitTime = Math.max(0, resetTime.getTime() - Date.now());
        
        if (waitTime > 0) {
          await new Promise(resolve => setTimeout(resolve, waitTime));
        } else {
          // Exponential backoff if reset time not provided
          await new Promise(resolve => setTimeout(resolve, delay));
          delay *= 2; // Double the delay for next retry
        }
        
        if (attempt < maxRetries - 1) continue;
      }
      
      return response;
    } catch (error) {
      if (attempt === maxRetries - 1) throw error;
      await new Promise(resolve => setTimeout(resolve, delay));
      delay *= 2;
    }
  }
}

Rate Limit Best Practices​

1. Monitor Rate Limit Headers​

class RateLimitMonitor {
  constructor() {
    this.limits = new Map();
  }
  
  update(url, response) {
    const limit = parseInt(response.headers.get('X-RateLimit-Limit') || '0');
    const remaining = parseInt(response.headers.get('X-RateLimit-Remaining') || '0');
    const reset = response.headers.get('X-RateLimit-Reset');
    
    this.limits.set(url, { limit, remaining, reset });
    
    // Warn when approaching limit
    if (remaining < limit * 0.1) {
      console.warn(`Rate limit warning for ${url}: ${remaining}/${limit} remaining`);
    }
  }
  
  shouldThrottle(url) {
    const limit = this.limits.get(url);
    if (!limit) return false;
    
    return limit.remaining === 0;
  }
}

2. Implement Request Queuing​

class RateLimitedQueue {
  constructor() {
    this.queue = [];
    this.processing = false;
  }
  
  async add(requestFn) {
    return new Promise((resolve, reject) => {
      this.queue.push({ requestFn, resolve, reject });
      this.process();
    });
  }
  
  async process() {
    if (this.processing || this.queue.length === 0) return;
    
    this.processing = true;
    
    while (this.queue.length > 0) {
      const { requestFn, resolve, reject } = this.queue.shift();
      
      try {
        const response = await requestFn();
        
        if (response.status === 429) {
          // Re-queue request
          this.queue.unshift({ requestFn, resolve, reject });
          const error = await response.json();
          const resetTime = new Date(error.resetAt);
          const waitTime = Math.max(0, resetTime.getTime() - Date.now());
          await new Promise(r => setTimeout(r, waitTime));
          continue;
        }
        
        resolve(response);
      } catch (error) {
        reject(error);
      }
    }
    
    this.processing = false;
  }
}

3. Cache Responses​

// Cache responses to reduce API calls
const cache = new Map();

async function fetchWithCache(url, options, ttl = 60000) {
  const cacheKey = `${url}:${JSON.stringify(options)}`;
  const cached = cache.get(cacheKey);
  
  if (cached && Date.now() - cached.timestamp < ttl) {
    return cached.response;
  }
  
  const response = await fetch(url, options);
  const data = await response.json();
  
  cache.set(cacheKey, {
    response: data,
    timestamp: Date.now()
  });
  
  return data;
}

Rate Limit Error Response​

When rate limited, you'll receive:

{
  "statusCode": 429,
  "message": [
    "Too many requests. Limit: 5 per 900 seconds"
  ],
  "limit": 5,
  "remaining": 0,
  "resetAt": "2025-01-15T10:15:00.000Z",
  "timestamp": "2025-01-15T10:00:00.000Z",
  "path": "/v1/auth/login"
}
